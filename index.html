How to use TheaterJS

  1. Include jQuery and TheatherJS

    <script src="js/jquery-1.7.1.min.js" type="text/javascript"></script>
<script src="js/jquery-ui-custom.min.js" type="text/javascript"></script>
<script src="js/imageteaser.js" type="text/javascript"></script>

  2. Ready your HTML

<head>
<script>
$(window).load(function() {
   $('.theater-js').theater({
      animationSpeed    : 1000,         
      animationEffect   : 'easeInOutExpo',      
      timer             : true,
      advanceSpeed      : 4000, 
      backgroundColor   : '#FFF'
   });
});
</script>
</head>
